CMD规范-seaJS
参考：https://seajs.github.io/seajs/docs/#docs
定义暴露模块:
define(function(require, module, exports){
  通过require引入依赖模块
  通过module/exports来暴露模块【与commonJS相同】
  exports.xxx = value
})
使用模块:
seajs.use(['模块1', '模块2'])

页面引用：
<script type="text/javascript" src="../../../js/sea.js"></script>

配置脚本：
// seajs 的简单配置
seajs.config({
  base: "../../../js",
  alias: {
    "jquery": "jquery-1.10.1.js"
  }
})

// 加载入口模块
seajs.use("./js/main")

main.js:
define(function(require, exports, module) {
  require('jquery');
  var show = require('./show')
  $('button').click(function(){
    show();
  })
});

message.js
define(function(require, exports, module) {
  require('jquery');
  exports.getMessage = function(){
    return $('#name').val();
  }
})

show.js
define(function(require, exports, module) {
  var msg = require('./message');
  module.exports = function(){
    alert(msg.getMessage())
  }
})
