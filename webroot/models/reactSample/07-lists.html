function PersonList(props){
	var personItems = props.persons.map((person,index) => <li key={person.no}>{person.name}</li>);
	return (<ul>{personItems}</ul>);
}

var persons = [{no:1,name:"丁怡"},{no:2,name:"卫子贤"},{no:3,name:"王牧"},{no:4,name:"卢平镇"}];
ReactDOM.render(
	<PersonList persons={persons}/>,
	document.getElementById("example")
);
