函数式定义无状态组件
页面：
<div id="example1"></div>

脚本：
function Hello(props){
	return <h1>您好，{props.name}！</h1>
}

ReactDOM.render(
	<Hello name="张三"/>,
	document.getElementById('example1')
);
React.createClass方式创建组件（ES5形式，旧版本功能，性能有缺陷，不建议用）
页面：
<div id="example2"></div>

脚本：
var Hello = React.createClass({
	render: function() {
        return (
            <h1>您好，{this.props.name}！</h1>
        );
    }
});
ReactDOM.render(
	<Hello name="李四"/>,
	document.getElementById('example2')
);
React.Component方式创建组件（ES6形式，推荐）
页面：
<div id="example3"></div>

脚本：
class Hello extends React.Component{
	constructor(props) {
		super(props);
	}
	render(){
        return (
            <h1>您好，{this.props.name}！</h1>
        );
    }
}
ReactDOM.render(
	<Hello name="王五"/>,
	document.getElementById('example3')
);
