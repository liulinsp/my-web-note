使用props.children构造组件
页面：
<div id="example1"></div>

脚本：
function Box(props){
	return (
		<div>
			<h3>{props.title}</h3>
			<div class="content">
				{props.children}
			</div>
		</div>
	)
}

ReactDOM.render(
	<Box title="提示：">
		<p>签到有红包！</p>
	</Box>,
	document.getElementById("example1")
);
使用自定义参数构造组件
页面：
<div id="example2"></div>

脚本：
function Box(props){
	return (
		<div>
			<h3>{props.title}</h3>
			<div class="content">
				{props.content}
			</div>
			<div class="option">
				{props.option}
			</div>
		</div>
	)
}

var content = (<p>确认退出系统？</p>);
var option = (<p><button>确定</button> <button>取消</button></p>);
ReactDOM.render(
	<Box title="警告：" content={content} option={option}/>,
	document.getElementById("example2")
);
