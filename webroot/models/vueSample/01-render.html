绑定 DOM 元素属性
页面：
<div id="app2">
	<span v-bind:title="mytitle">鼠标放到我上面你就会看到动态绑定的标题</span>
</div>

脚本：
var app2 = new Vue({
	el:"#app2",
	data:{
		mytitle:"这是绑定的标题"
	}
});

简写：
<span :title="mytitle">鼠标放到我上面你就会看到动态绑定的标题</span>

循环渲染
页面：
<div id="app4">
  <li v-for="item in myList">{{item.name}}</li>
</div>

脚本：
var app4 = new Vue({
	el:"#app4",
	data:{
		myList:[
			{name:"aaa"},
			{name:"bbb"},
			{name:"ccc"}
		]
	}
});
  • {{item.name}}
