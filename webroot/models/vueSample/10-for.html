基本用法
页面：
<div id="app1">
  <li v-for="item in myList">{{item.name}}</li>
</div>

脚本：
var app1 = new Vue({
	el:"#app1",
	data:{
		myList:[
			{name:"aaa"},
			{name:"bbb"},
			{name:"ccc"}
		]
	}
});
  • {{item.name}}
    • 显示索引
    页面：
<div id="app2">
	<li v-for="(item,index) in myList">{{index}} - {{item.name}}</li>
</div>

脚本：
var app2 = new Vue({
	el:"#app2",
	data:{
		myList:[
			{name:"aaa"},
			{name:"bbb"},
			{name:"ccc"}
		]
	}
});
  • {{index}} - {{item.name}}
    • v-for使用template标签
    如同 v-if 模板，你也可以用带有 v-for 的 <template> 标签来渲染多个元素块。
最终的渲染结果不会包含template标签。

页面：
<div id="app3">
<template v-for="(item,index) in myList">
	<li>{{index}} - {{item.name}}</li>
</template>
</div>


脚本：
var app3 = new Vue({
	el:"#app3",
	data:{
		myList:[
			{name:"aaa"},
			{name:"bbb"},
			{name:"ccc"}
		]
	}
});

渲染后结果:
<div id="app3">
	<li>0 - aaa</li>
	<li>1 - bbb</li>
	<li>2 - ccc</li>
</div>
    对象迭代
    页面：
<div id="app4">
	<li v-for="(value, key, index) in myObj">{{index}}、{{key}} ：{{value}}</li>
</div>

脚本：
var app4 = new Vue({
	el:"#app4",
	data:{
		myObj:{
			name:"liulin",
			age:18,
			sex:"woman"
		}
	}
});
  • {{index}}、{{key}} ：{{value}}
    • key
    页面：
<div id="app5">
	<div v-for="(item, index) in myList" v-bind:key="item.id">
		{{index}} ：{{item.name}} <input type="text">
	</div>
	<button @click="btnHandler">倒序</button>
</div>

脚本：
var app5 = new Vue({
	el:"#app5",
	data:{
		myList:[
			{id:111,name:"张三"},
			{id:222,name:"李四"},
			{id:333,name:"王五"}
		]
	},
	methods:{
		btnHandler:function(){
			this.myList.reverse();
		}
	}
});
    {{index}} ：{{item.name}}
    显示过滤结果
    页面：
<div id="app6">
	<div>
	    原数字列表：
		<li v-for="n in numbers">{{n}}</li>
	</div>

	<div>
		奇数列表：
		<li v-for="n in evenNumbers">{{n}}</li>
	</div>
</div>

脚本：
var app6 = new Vue({
	el:"#app6",
	data:{
		numbers:[1,2,3,4,5]
	},
	computed:{
		evenNumbers:function(){
			return this.numbers.filter(function(num){
				return num%2;
			});
		}
	}
});
    原数字列表：
  • {{n}}
    • 奇数列表：
  • {{n}}
