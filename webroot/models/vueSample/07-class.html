对象语法
样式：
.my-item{
	width: 200px;
	height: 50px;
	margin: 10px;
	border: 1px solid #f1f1f1;
	box-sizing: border-box;
}
.active{
	border: 2px solid #A3FEB0;
}

页面：
<div id="app1">
    <div :class="singleClassObj">单独对象</div>

    列表：
	<div class="my-item" v-for="(item,index) in items" 
	     v-bind:class="{active: item.isActive}">列表：{{index}}</div>
</div>

脚本：
var app1 = new Vue({
	el:"#app1",
	data:{
		singleClassObj:{
			"my-item":true,
			active:true
		},
		items:[{isActive:false},{isActive:true},{isActive:false}]
	}
});
