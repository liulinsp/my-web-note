javascript过渡
引入第三方脚本库：
<script type="text/javascript" src="../../js/velocity.js"></script>

页面：
<div id="app3">
	<transition 
		appear
		v-on:before-appear="beforeAppear"
		v-on:appear="appear"
		v-on:after-appear="afterAppear"
		v-on:appear-cancelled="appearCancelled"
		v-bind:css="false">
		<div>你好！</div>
	</transition>	
</div>

脚本：
var app3 = new Vue({
	el:"#app3",
	created () {
		this.begin = Date.now()
	},
	methods: {
	    beforeAppear: function (el) {
	      console.log('beforeAppear', Date.now() - this.begin);
	      el.style.opacity = 0
	      el.style.transformOrigin = 'left'
	    },
	    appear: function (el, done) {
	      console.log('appear', Date.now() - this.begin);
	      Velocity(el, { opacity: 1, fontSize: '1.4em',  translateX:'50px'}, { duration: 500 })
	      Velocity(el, { fontSize: '1em',  translateX:'0' }, { duration: 200, complete: done })
	    },
	    afterAppear: function (el, done) {
	      console.log('afterAppear', Date.now() - this.begin);
	    },
	    appearCancelled: function (el, done) {
	      console.log('appearCancelled', Date.now() - this.begin);
	    }
	}
});
你好！
