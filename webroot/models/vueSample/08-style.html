对象语法
页面：
<div id="app1">
	<div v-bind:style="curStyle">你好，{{curUser.name}}</div>
</div>

脚本：
var app1 = new Vue({
	el:"#app1",
	data:{
		curUser:{name:"琳",isVIP:true}
	},
	computed:{
		curStyle:function(){
			if(this.curUser.isVIP){
				return {
					color:"red"
				};
			}else{
				return {
					color:"gray"
				};
			}
		}
	}
});
你好，{{curUser.name}}

<div id="app2">
	<div v-bind:style="[fontStyles,borderStyles]">你好!</div>
</div>

脚本：
var app2 = new Vue({
	el:"#app2",
	data:{
		fontStyles:{
			color:"red",
			fontSize:"20px",
			fontWeight:700
		},
		borderStyles:{
			borderWidth:"2px",
			borderColor:"red",
			borderStyle:"solid"
		}
	}
});
