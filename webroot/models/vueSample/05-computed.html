计算属性
页面：
<div id="app1">
	{{reverseMessage}}
</div>

脚本：
var app1 = new Vue({
	el:"#app1",
	data:{
		message:"123456789"
	},
	computed:{
		reverseMessage:function(){
			return this.message.split('').reverse().join('');
		}
	}
});

计算属性set、get
页面：
<div id="app2">
	<h3>账号：{{account}}</h3>
	<h3>后缀：{{suffix}}</h3>
	<input type="text" v-model="email">
	<h3>邮箱：{{email}}</h3>
</div>

脚本：
var app2 = new Vue({
	el:"#app2",
	data:{
		account:"linliu",
		suffix:"sina.com"
	},
	computed:{
		email:{
			set:function(newVal){
				var valArray = newVal.split("@");
				if(valArray.length>0){
					this.account=valArray[0];
				}else{
					this.account="";
				}

				if(valArray.length>1){
					this.suffix=valArray[1];
				}else{
					this.suffix="";
				}
			},
			get:function(){
				if(this.suffix!=""){
					return this.account+"@"+this.suffix;
				}else{
					return this.account;
				}
			}
		}
	}
});

账号：{{account}}

后缀：{{suffix}}

邮箱：{{email}}
