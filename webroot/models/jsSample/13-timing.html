停止计时
<button onclick="startMsg();">开始</button>
<button onclick="stopMsg();">停止</button>

脚本：
var timer;
function startMsg (){
	$('<div>2秒到时了</div>').appendTo($("#showResult3"));
	timer = setTimeout("startMsg()", 2000);
}

function stopMsg(){
	clearTimeout(timer);
}
停止循环
<button onclick="startInterval();">开始</button>
<button onclick="stopInterval();">停止</button>
脚本：
var timer;
function startInterval (){
	timer = setInterval(function(){
		$('<div>2秒到时了</div>').appendTo($("#showResult5"));
	}, 2000);
}

function stopInterval(){
	clearInterval(timer);
}
