in操作符，会遍历原型链
function Person(name){
	this.name = name;
}	
Person.prototype.leg = 2;

var p = new Person('Tom');

'leg' in p ; // true
'name' in p ; // true
obj.hasOwnProperty('propName')，只检查对象本身
function Person(name){
	this.name = name;
}	
Person.prototype.leg = 2;

var p = new Person('Tom');

p.hasOwnProperty('leg') ; // false
p.hasOwnProperty('name'); // true
