定义函数
脚本：
var obj1 ={
	name:"lin",
	hello:function(){
		$("#showResult1").append("<p>你好,"+this.name+"</p>");
	}
};
//执行
obj1.hello();

/*ES6新特性*/
var obj2 ={
	name:"ES6",
	hello(){
		$("#showResult1").append("<p>你好,"+this.name+"</p>");
	}
};
//执行
obj2.hello();
箭头函数
脚本：
var obj1 ={
	hello:function(name,sex){
		$("#showResult2").append("<p>你好,"+name+sex+"</p>");
	}
};
//执行
obj1.hello("lin","女士");

/*ES6新特性*/
var obj2 ={
	hello: (name,sex) => $("#showResult2").append("<p>你好,"+name+sex+"</p>")
};
//执行
obj2.hello("ES6","先生");
