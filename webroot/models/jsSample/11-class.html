//class从ES6开始正式被引入到JavaScript中的，有些浏览器不支持，例如IE
class Student {
    constructor(name) {
        this.name = name;
    }

    hello() {
        alert('Hello, ' + this.name + '!');
    }
}

class PrimaryStudent extends Student {
    constructor(name, grade) {
        super(name); // 记得用super调用父类的构造方法!
        this.grade = grade;
    }

    myGrade() {
        alert('I am at grade ' + this.grade);
    }
}

var zhang = new PrimaryStudent("zhang san",3)
zhang.hello();
zhang.myGrade();
