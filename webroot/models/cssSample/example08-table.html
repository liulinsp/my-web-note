脚本：
  $("#demoTable1>tbody>tr").each(function(){
    $(this).hover(handlerIn, handlerOut);
  });

  function handlerIn() {
    $(this).find('td').css("border-width","0px");
    $(this).prev("tr").find('td').css("border-bottom-width","0px");
    if($(this).prev("tr").length==0){
      $(this).parent().prev("thead").find("tr")
        .css("border-bottom-width","0px");
    }
    $(this).next("tr").find('td').css("border-top-width","0px");
    $(this).css({
      "border-style": "solid",
      "border-width": "1px",
      "border-color": "#484EE3"
    });
  }

  function handlerOut() {
    $(this).find('td').css("border-width","1px");
    $(this).prev("tr").find('td').css("border-bottom-width","1px");
    if($(this).prev("tr").length==0){
      $(this).parent().prev("thead").find("tr")
        .css("border-bottom-width","1px");
    }
    $(this).next("tr").find('td').css("border-top-width","1px");
    $(this).css({
      "border-width": "0px",
    });
  }
