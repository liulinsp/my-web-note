clip:rect矩形剪裁只能作用于position:absolute的元素上
/*显示第一行第一列图标*/
.img1-1{
	position:absolute;
	margin-top: -40px;
	margin-left: -40px;
	clip: rect(40px 120px 120px 40px);
}
		
/*显示第一行第二列图标*/
.img1-2{
	position:absolute;
	margin-top: -40px;
	margin-left: -160px;
	clip: rect(40px 240px 160px 120px);
}
		
/*显示第二行第一列图标*/
.img2-1{
	position:absolute;
	margin-top: -160px;
	margin-left: -40px;
	clip: rect(160px 120px 240px 40px);
}
		
/*显示第二行第二列图标*/
.img2-2{
	position:absolute;
	margin-top: -160px;
	margin-left: -160px;
	clip: rect(160px 240px 240px 160px);
}
