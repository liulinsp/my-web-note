改变光标颜色
input::-webkit-input-placeholder{ 
	color: blue; /* 改变placeholder文本颜色 */ 
	text-shadow: none;
	-webkit-text-fill-color: initial; 
}

/* firefox48.0以下不支持，IE任何版本都不支持 */
@supports (-webkit-text-fill-color: transparent) { 
	input, textarea {
		/* 文字和光标颜色，由于文字镂空显示阴影的颜色，只有光标的颜色为红色 */
		color: red !important;
		text-shadow: 0px 0px 0px #495057;/* 阴影为文字大小 */
		-webkit-text-fill-color: transparent;/*让文本变成镂空的*/
	}
}

/* IE、Safari任何版本都不支持 */
@supports (caret-color: red) { 
	input, textarea{ 
		color: #495057; /* 文本颜色 */ 
		caret-color: red; /* 光标颜色 */ 
	} 
}
