全局作用域
$scope 是一个 JavaScript 对象，带有属性和方法，
  这些属性和方法可以在视图和控制器中使用。

$rootScope 可作用于整个应用中。
  是各个 controller 中 scope 的桥梁。
  用 rootscope 定义的值，可以在各个 controller 中使用。

<div ng-app="app1" >
  <div ng-controller="ctrl1" style="background-color:#C6FCD7">
    <li ng-repeat="x in names">{{country+'-'+city+'-'+x}}</li>
  </div>
  <div ng-controller="ctrl2" style="background-color:#FCB3EB">
    <li ng-repeat="x in names">{{country+'-'+city+'-'+x}}</li>
  </div>
</div>

脚本：
var app1 = angular.module("app1",[]);
app1.controller("ctrl1",function($scope, $rootScope){
  $scope.names=["张三","李四","王五"];
  $scope.city="北京";
  $rootScope.country = "中国";
});
app1.controller("ctrl2",function($scope, $rootScope){
  $scope.names=["赵佳","刘丽","杨洋"];
  $scope.city="天津";
});
