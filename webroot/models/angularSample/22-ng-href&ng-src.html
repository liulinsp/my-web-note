ng-href
<div id="app1" ng-controller="ctrl1">
  <p>
    <a href="{{data.myHref}}">{{data.title}}</a>
    (超链接在myHref初始前是可点的)
  </p>
  <p>
    <a ng-href="{{data.myHref}}">{{data.title}}</a>
    (超链接在myHref初始前是不可点的)
  </p>
</div>

脚本:
angular.module("app1",[])
  .controller("ctrl1",function($scope,$timeout){
    $scope.data={title:"未设置myHref"}
    $timeout(function(){
      $scope.data.myHref="http://www.baidu.com";
      $scope.data.title="已设置myHref";
    },5000);
  });
angular.bootstrap(document.getElementById("app1"),["app1"]);

{{data.title}} (超链接在myHref初始前是可点的)

{{data.title}} (超链接在myHref初始前是不可点的)
