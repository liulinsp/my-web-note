返回字符串示例
<div id="app2">
  <span my-icon>这里会换成图片</span>
</div>

脚本：
angular.module("app2",[])
  .directive('myIcon',function(){
    return {
      restrict: 'A',
      replace:true,
      template:"<img src='../../img/arrow_cur_l.png'/>"
    };
  });
angular.bootstrap(document.getElementById("app2"), ['app2']);
这里会换成图片
返回函数示例
<div id="app1" ng-controller="ctrl1">
  <span my-input="user.name"></span>
  {{user.name}}
</div>

angular.module("app1",[])
  .directive('myInput',function(){
    return {
      restrict: 'A',
      replace:true,
      template:function(element,attr){
        return "<input type='text' ng-model='"+attr.myInput+"' />";
      }
    };
  })
  .controller("ctrl1",function($scope){
    $scope.user={name:"张三"};
  });
angular.bootstrap(document.getElementById("app1"), ['app1']);
{{user.name}}
templateUrl
templateUrl是可选的参数，可以是以下类型：
  1. 一个代表外部HTML文件路径的字符串；
  2. 一个可以接受两个参数的函数，参数为tElement和tAttrs，
      并返回一个外部HTML文件路径的字符串。
