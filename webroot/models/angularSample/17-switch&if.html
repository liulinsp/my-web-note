ng-switch
<div id="app1" ng-controller="ctrl1">
  <div ng-repeat="user in users" ng-switch on="user.type">
    <p ng-switch-default>{{ user.name }} (会员卡)</p>
    <p ng-switch-when="silver">{{ user.name }}(银卡)</p>
    <p ng-switch-when="gold">{{ user.name }}(金卡)</p>
  </div>      
</div>

脚本：
var app1 = angular.module("app1",[]);
app1.controller("ctrl1",function($scope){
  $scope.users=[{name:"李音",type:"silver"},
                {name:"王萌",type:""},
                {name:"张晋",type:"gold"}];
});
angular.bootstrap(document.getElementById("app1"), ['app1']);

<div id="app2" ng-controller="ctrl2">
  <div ng-repeat="user in users" ng-if="user.type=='gold'">
    <p>{{ user.name }}(金卡)</p>
  </div>      
</div>
