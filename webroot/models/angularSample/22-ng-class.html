1. 对象key/value处理
used==true并且vip==true的用户才高亮显示
<div id="app1" ng-controller="ctrl1">
  <div ng-repeat="u in users" 
       ng-class="{'height-light':u.used&&u.vip}">
    {{u.name}}[used:{{u.used}},vip:{{u.vip}}]
  </div>
</div>

脚本：
angular.module("app1",[])
  .controller("ctrl1",function($scope){
    $scope.users=[{name:"正在使用的VIP用户",used:true,vip:true},
                  {name:"已停用的VIP用户",used:false,vip:true},
                  {name:"正在使用的普通用户",used:true,vip:false}];
  });
angular.bootstrap(document.getElementById("app1"),["app1"]);
样式：
.height-light{
  background-color: yellow;
}
{{u.name}}[used:{{u.used}},vip:{{u.vip}}]
2. 字符串数组形式
<div id="app2" ng-controller="ctrl2">
  <div ng-repeat="u in users" 
       ng-class="{true:'vip',false:'normal'}[u.vip]">
    {{u.name}}[vip:{{u.vip}}]
  </div>
</div>

脚本：
angular.module("app2",[])
  .controller("ctrl2",function($scope){
    $scope.users=[{name:"正在使用的VIP用户",vip:true},
                  {name:"已停用的VIP用户",vip:true},
                  {name:"正在使用的普通用户",vip:false}];
  });
angular.bootstrap(document.getElementById("app2"),["app2"]);
样式：
.vip{
  background-color: yellow;
}
.normal{
  background-color: gray;
}
{{u.name}}[vip:{{u.vip}}]
