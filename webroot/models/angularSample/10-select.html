简单示例
<div ng-app="app1" ng-controller="ctrl1">
  <select ng-model="selectedClass" ng-options="x for x in classes"></select>
</div>

脚本：
var app1 = angular.module("app1",[]);
app1.controller("ctrl1",function($scope){
  $scope.classes=["宅男","剁手党","高品质"];
  $scope.selectedClass="高品质";
});
ng-options方式含value示例
<div ng-app="app4" ng-controller="ctrl4">
    <select name="tag" ng-model="tag" ng-options="x.val as x.name for x in tags">
    </select>
</div>

脚本：
var app4 = angular.module("app4",[]);
app4.controller("ctrl4",function($scope){
  $scope.tags=[{val:"01",name:"宅男"},{val:"02",name:"剁手党"},{val:"03",name:"高品质"}];
  $scope.tag="03";
});
ng-repeat方式含value示例
<div ng-app="app2" ng-controller="ctrl2">
    <select name="tag" ng-model="tag" >
      <option value="">请选择</option>
      <option ng-repeat="x in tags" value="{{x.val}}">{{x.name}}</option>
    </select>
</div>

脚本：
var app2 = angular.module("app2",[]);
app2.controller("ctrl2",function($scope){
  $scope.tags=[{val:"01",name:"宅男"},{val:"02",name:"剁手党"},{val:"03",name:"高品质"}];
  $scope.tag="03";
});
ng-repeat方式多选示例
<div ng-app="app3" ng-controller="ctrl3">
    <select name="tag" ng-model="tag" multiple="multiple">
      <option value="">请选择</option>
      <option ng-repeat="x in tags" value="{{x.val}}">{{x.name}}</option>
    </select>
    <p>您选择的是：<li ng-repeat="x in tag">{{x}}</li></p>
</div>

脚本：
var app3 = angular.module("app3",[]);
app3.controller("ctrl3",function($scope){
  $scope.tags=[{val:"01",name:"宅男"},{val:"02",name:"剁手党"},{val:"03",name:"高品质"}];
  $scope.tag=["01","03"];
});
