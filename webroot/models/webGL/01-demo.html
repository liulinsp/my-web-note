绘制简单图形
页面：
<div class="aShow note" id="app1"></div>

脚本：
	// HTML中已有canvas标签
	/* var renderer = new THREE.WebGLRenderer({
	    canvas: document.getElementById('canvas1')
	}); */

	// HTML中无canvas标签，自动创建
	var renderer = new THREE.WebGLRenderer({antialias:true}); // antialias: true 抗锯齿
	renderer.setSize(400, 300);
	var parentEle = document.getElementById('app1')
	parentEle.appendChild(renderer.domElement);

	// 将背景色（用于清除画面的颜色）设置为黑色
	renderer.setClearColor(0x000000);
	// 场景
	var scene = new THREE.Scene();
	// 相机
	var camera = new THREE.PerspectiveCamera(45, 4 / 3, 1, 1000);
	camera.position.set(0, 0, 5); // 相机位置
	camera.lookAt(new THREE.Vector3(0, 0, 0));
	scene.add(camera);

	// 材料
	var material = new THREE.MeshBasicMaterial({
	        color: 0xffffff // white
	});

	// 创建x、y、z方向长度分别为1、2、3的长方体
	var cube = new THREE.Mesh(new THREE.CubeGeometry(1, 2, 3), material);
	scene.add(cube); // 长方体添加到场景中

	// 进行渲染
	renderer.render(scene, camera);
