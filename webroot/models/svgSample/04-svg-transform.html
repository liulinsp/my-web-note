<script type="text/javascript" src="../../js/svg-morpheus.js"></script>

var weatherIcon = new SVGMorpheus('#weather');
var curWeatherIcon = ''

function changeWeather () {
	curWeatherIcon = curWeatherIcon === 'cloud' ? 'lightning' : 'cloud';
	// weatherIcon.to(curWeatherIcon, {duration: 300, easing: 'quad-in-out', rotation: 'none'});
	weatherIcon.to(curWeatherIcon, {duration: 500, easing: 'elastic-out', rotation: 'none'});
}

changeWeather();
setInterval(changeWeather, 2000);

脚本：
<script type="text/javascript" src="../../js/svg-morpheus.js"></script>

var svgIcon = new SVGMorpheus('#icon');
var curIcon = ''

function change () {
	curIcon = curIcon === 'star' ? 'heart' : 'star';
	svgIcon.to(curIcon, {duration: 500, easing: 'quad-in-out', rotation: 'clock'});
}

change();
setInterval(change, 2000);
